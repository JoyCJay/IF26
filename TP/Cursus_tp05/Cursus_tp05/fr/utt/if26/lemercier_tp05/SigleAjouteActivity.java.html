package fr.utt.if26.lemercier_tp05; 
 
import android.content.Intent; 
import android.os.Bundle; 
import android.support.v7.app.AppCompatActivity; 
import android.util.Log; 
import android.view.View; 
import android.widget.Button; 
import android.widget.EditText; 
 
public class SigleAjouteActivity extends AppCompatActivity implements View.OnClickListener { 
 
    EditText et; 
    Button bt; 
 
 
    @Override 
    protected void onCreate(Bundle savedInstanceState) { 
        super.onCreate(savedInstanceState); 
        setContentView(R.layout.activity_sigle_ajoute); 
        et = findViewById(R.id.sigle_ajoute_et); 
        bt = findViewById(R.id.sigle_ajoute_bt); 
 
        bt.setOnClickListener(this); 
    } 
 
    @Override 
    public void onClick(View v) { 
 
        String sigle = (String) et.getText().toString(); 
 
 
        MonApplication.listeAjoute(sigle); 
        MonApplication.listeAjoute("test"); 
 
        Log.i ("Ajoute d'un sigle", sigle); 
 
        Intent intent = new Intent (this, SigleListeActivity.class); 
        startActivity(intent); 
 
        // Comment faire pour ajouter le nouveau sigle dans la liste définie dans une autre activity ? 
 
        /* 
 
        https://stackoverflow.com/questions/33281674/how-to-access-the-same-variable-between-activities-in-android 
 
        Solution 1 
        Toutes les activités dérivent d'une BaseActivity 
        Ajoutez vos variable globales en mode protected et STATIC 
        Pas de besoin de singleton 
 
        Solution 2 
        Créer une classe singleton 
 
 
        Ajouter vos STATIC variables et les get/Set 
 
 
 
 
        Solution 3 
        Creer une classe MonApplication dérivant de Application 
            Ajouter des variable static 
 
        --> ne fonctionne pas car pas possible de faire des ajout dans une list static .... 
 
 
        Solution 4 
            Utiliser SharedPreference 
 
 
        */ 
 
    } 
}
